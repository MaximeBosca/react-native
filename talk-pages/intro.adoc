=== !
[source,tsx]
--
import { StyleSheet } from 'react-native'; // important

const styles = StyleSheet.create({
    container: {
        backgroundColor: 'red',
        marginBottom: 20
    },
    text: {
        color: 'blue'
    }
})
--
[source,tsx]
--
<View style={styles.container}>
    <Text style={styles.text}>Ce texte est bleu sur fond rouge !</Text>
</View>
